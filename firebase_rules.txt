rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Shared state per customer
    match /sessions/{customer} {
      allow read: if true;
      allow write: if request.auth != null; // anonymous users can write session state
    }

    // Presence (who's online) â€” doc id MUST equal the user's UID
    match /sessions/{customer}/presence/{uid} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.uid == uid;
    }
  }
}
